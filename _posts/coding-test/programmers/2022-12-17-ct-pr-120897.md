import java.util.HashSet;
import java.util.Set;

class Solution {
    public int[] solution(int n) {
        Set<Integer> answer = new HashSet<>();
        for(int i = 1; i <= Math.sqrt(n); i++){
            if(n%i == 0){
                answer.add(i);
                answer.add(n/i);
            }
        }
        return answer.stream().sorted().mapToInt(i->i).toArray();
    }
}
