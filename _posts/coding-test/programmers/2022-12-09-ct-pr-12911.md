class Solution {
    public int solution(int n) {
        int answer = 0;
        int count = getOneCount(n);
        while(true){
            if(getOneCount(++n) == count){
                answer = n;
                break;
            }
        }
        return answer;
    }
    public int getOneCount(int n){
        int count = 0;
        while(n != 0){
            if(n%2 == 1)
                count++;
            n/=2;
        }
        return count;
    }
}
