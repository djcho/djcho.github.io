using namespace std;

int solution(int n) {
    vector<int> vec;
    while(n > 2){
        vec.push_back(n%3);
        n /= 3;
    }
    vec.push_back(n);

    int sum = 0;
    int vec_size = vec.size() - 1;
    for(int i = 0; i < vec.size(); i++){
        sum += vec[i] * pow(3, vec_size);
        vec_size--;
    }
    
    return sum;
}
