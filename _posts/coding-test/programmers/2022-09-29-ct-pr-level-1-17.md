vector<int> solution(vector<int> arr, int divisor) {
    vector<int> answer;
    for(auto& i : arr){
        if(i%divisor == 0){
            answer.push_back(i);
        }        
    }
    if(answer.empty())
        return {-1};

    std::sort(answer.begin(), answer.end());
    return answer;
}
